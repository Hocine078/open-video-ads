Example 29: Modifying the Linear Ad Clickthrough Message

You can modify the font style, size and whether or not the message is shown.


The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.5.swf", {
    plugins: {
        controls: {
            autoHide: "always"
        },
        rtmp: {
            url: "../dist/flowplayer.rtmp-3.1.3.swf"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "debug": {
                "debugger": "firebug",
                "levels": "fatal"
            },

            "ads": {
                "pauseOnClickThrough": true,
                "clickSign": {
                    "enabled": true, 
                    "verticalAlign": "center",
                    "horizontalAlign": "center",
                    "width": 150,
                    "height": 32,
                    "opacity": 0.5,
                    "borderRadius": 20,
                    "backgroundColor": "#000000",
                    "style": ".smalltext { font-size:12; }",
                    "html": "<p class=\"smalltext\" align=\"center\">CLICK ME!</p>",
                    "scaleRate": 0.75
                },
                "notice": { "show": true },
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php",
                       "customProperties": { 
                            "target": "category=food" 
                       }
                    }
                ],
                "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll"
                    }
                ]
            }
        }
    }
});
