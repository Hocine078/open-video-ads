Example 20 - Targetted Ad Delivery

This example delivers only ads that match a "food" category. To achieve this, ads are configured on the OpenX server with a "site variable" called "category" and a value of "food" is assigned.



The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/rtmp20.xml"
/>

"rtmp20.xml" is defined as: 


<config>
    <openadstreamer.title>
        RTMP Example 20
    </openadstreamer.title>
    <openadstreamer.json>
        {
            "allowPlaylistControl": true,

            "debug": {
                "debugger": "firebug",
                "levels": "fatal"
            },

            "ads": {
                "pauseOnClickThrough": true,
                "notice": { "textStyle": "smalltext" },
                "server": {
                    "type": "OpenX",
                    "adParameters": [ { "name": "category", "value": "food" } ],
                    "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                },
                "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll",
                    }
                ]
            }
        }
    </openadstreamer.json>
</config>
