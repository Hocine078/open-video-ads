Example 58 - Impression serving on an empty Ad VAST response

This example illustrates how an ad server configuration can be set to ensure that the impression URL is called even if the VAST response provides no ad.


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/all58.xml"
/>

"all58.xml" is defined as: 


<config>
    <openadstreamer.title>
       ALL Example 26
    </openadstreamer.title>
    <openadstreamer.json>
       {
            "allowPlaylistControl": true,

            "debug": {
                "debugger": "firebug",
                "levels": "fatal"
            },

            "shows": {
                "baseURL": "http://streaming.openvideoads.org:81/shows",
                "streams": [
                    { "file":"the-black-hole.mp4", "duration":"00:01:00" },
                    { "file":"the-black-hole.mp4", "duration":"00:01:00" }
                ]
            },

            "ads": {
                "servers": [
                    {
                       "type": "direct",
                       "forceImpressionServing": true,
                       "apiAddress": "http://static.openvideoads.org/tests/empty-ad-vast-response.xml"
                    }
                ],

                "schedule": [
                    {
                       "zone": "18",
                       "position": "pre-roll",
                       "applyToParts": [ 0 ]
                    },
                    {
                       "zone": "18",
                       "position": "post-roll"
                    }
               ]
            }
        }
    </openadstreamer.json>
</config>
