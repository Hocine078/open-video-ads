Example 51 (ALL) - AdTech call returning XML wrapper to OpenX

This example illustrates a call to AdTech which returns an XML wrapped call to OpenX to fulfill the ad request


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/all51.xml"
/>

"all51.xml" is defined as: 


<config>
    <openadstreamer.title>
       ALL Example 51
    </openadstreamer.title>
    <openadstreamer.json>
        {
            "debug": {
                "debugger": "firebug",
                "levels": "fatal"
            },

            "ads": {
                "pauseOnClickThrough": true,
                "servers": [
                    {
                        "type": "AdTech",
                        "apiAddress": "http://de.at.atwola.com/?adrawdata/3.0/515.1"
                    }
                ],
                "schedule": [
                    {
                       "zone": "2189418/0/1725", 
                       "position": "pre-roll"
                    }
                 ]
             }
         }
    </openadstreamer.json>
</config>
