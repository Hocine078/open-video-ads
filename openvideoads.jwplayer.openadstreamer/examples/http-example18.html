Example 18 (HTTP)- Click on a linear ad to go to the advertiser website

Mouse over the ad and then click on it to be redirected to the advertiser website.



The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/http18.xml"
/>

"http18.xml" is defined as: 


<config>
   <openadstreamer.title>
        HTTP Example 18
   </openadstreamer.title>
   <openadstreamer.json>
        {
            "allowPlaylistControl": true,

            "debug": {
                "debugger": "firebug",
                "levels": "fatal"
            },

            "ads": {
                "pauseOnClickThrough": true,
                "notice": { "textStyle": "smalltext" },
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                    }
                ],
                "schedule": [
                    {
                       "zone": "21",
                       "position": "pre-roll"
                    }
                ]
            }
        }
   </openadstreamer.json>
</config>
