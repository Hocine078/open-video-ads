Example 15 - Selection of media based on bitrate

This example illustrates how to specify a 'bitrate' as a selection criteria for the mediafiles returned by the openX ad server.


The configuration for this example is: 


<object id="ply" width="450" height="300" type="application/x-shockwave-flash" name="ply" data="../dist/4.5.swf">
  <param name="allowfullscreen" value="true"/>
  <param name="allowscriptaccess" value="always"/>
  <param name="flashvars" value="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/rtmp15.xml"/>
</object>

"rtmp15.xml" is defined as: 


<config>
   <openadstreamer.title>
       RTMP Example 15
   </openadstreamer.title>
   <openadstreamer.json>
       {
            "deliveryType": "streaming",
            "allowPlaylistControl": false,

            "shows": {
                "baseURL": "rtmp://ne7c0nwbit.rtmphost.com/videoplayer",
                "streams": [
                    { "file":"the-black-hole.mp4", "duration":"00:00:15" }
                ]
            },

            "ads": {
                "bitrate": "500-700",
                "notice": { "textStyle": "smalltext" },
                "server": {
                    "type": "OpenX",
                    "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                },
                "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll"
                    }
                ]
            },

            "debug": {
                "levels": "fatal",
                "debugger": "firebug"
            }
        }
   </openadstreamer.json>
</config>
