Example 1 - A pre-roll ad only

This example plays an RTMP pre-roll ad only. It's defined as zone "1" in our OpenX instance.


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=350&width=660&plugins=../dist/OpenAdStreamer.swf&config=config/rtmp01.xml"
/>

"rtmp01.xml" is defined as: 


<config>
   <openadstreamer.title>
       RTMP Example 01
   </openadstreamer.title>
   <openadstreamer.json>
       {
            "allowPlaylistControl": true,

            "debug": {
                "debugger": "firebug",
                "levels": "fatal"
            },

            "ads": {
                "pauseOnClickThrough": true,
                "notice": { "textStyle": "smalltext" },
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                    }
                ],
                "schedule": [
                    {
                        "zone": "1",
                        "position": "pre-roll"
                    }
                ]
            }
       }
   </openadstreamer.json>
</config>
