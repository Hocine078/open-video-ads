Example 10 - A mixed RTMP and HTTP playlist

This example illustrates how to serve up a mixture of HTTP and RTMP streams (shows and ads). Zone '20' is a zone that serves all types of ads (RTMP and HTTP). This example plays a pre-roll ad (could be either HTTP or RTMP) and then a show stream (RTMP) followed by another pre-roll ad of some type and then a HTTP version of the show stream (each limited to 10 seconds).


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/all10.xml"
/>

"all10.xml" is defined as: 


<config>
   <openadstreamer.title>
       All Example 10
   </openadstreamer.title>
   <openadstreamer.json>
       {
            "allowPlaylistControl": false,

            "shows": {
                "streams": [
                   {
                     "file":"rtmp://ne7c0nwbit.rtmphost.com/videoplayer/the-black-hole.mp4",
                     "duration":"00:00:10"
                   },
                   {
                     "file":"http://streaming.openvideoads.org:81/shows/the-black-hole.mp4",
                     "duration":"00:00:10"
                   }
                ]
            },

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                    }
                ],
                "schedule": [
                   {
                      "zone": "22",
                      "position": "pre-roll"
                   }
                ]
            },

            "debug": {
                "levels": "fatal",
                "debugger": "firebug"
            }
        }
   </openadstreamer.json>
</config>
