Example 26 - Empty VAST Response

This example tests what happens when an empty VAST template is returned. The ad schedule should be skipped and the main stream(s) played.


The configuration for this example is: 


<object id="ply" width="450" height="300" type="application/x-shockwave-flash" name="ply" data="../dist/4.5.swf">
  <param name="allowfullscreen" value="true"/>
  <param name="allowscriptaccess" value="always"/>
  <param name="flashvars" value="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/all26.xml"/>
</object>

"http28.xml" is defined as: 


<config>
    <openadstreamer.title>
       ALL Example 26
    </openadstreamer.title>
    <openadstreamer.json>
       {
            "deliveryType": "progressive",
            "allowPlaylistControl": true,

            "debug": {
                "debugger": "firebug",
                "levels": "fatal"
            },

            "shows": {
                "baseURL": "http://streaming.openvideoads.org:81/shows",
                "streams": [
                    { "file":"the-black-hole.mp4", "duration":"00:01:00" },
                    { "file":"the-black-hole.mp4", "duration":"00:01:00" }
                ]
            },

            "ads": {
                "server": {
                "type": "OpenX",
                "vastURL": "http://streaming.openvideoads.org/tests/blank-vast-response.xml"
            },

            "schedule": [
                {
                    "zone": "18",
                    "position": "pre-roll",
                    "applyToParts": [ 0 ]
                },
                {
                    "zone": "18",
                    "position": "post-roll"
                }
            ]
          }
        }
    </openadstreamer.json>
</config>
