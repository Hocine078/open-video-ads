Example 2a - A pre-roll ad and show stream - No Playlist Control

This example plays a RTMP pre-roll ad before the main show stream. The ad slot is defined as zone "1" in our OpenX instance. The ad is repeated every time the stream is replayed. "allowPlaylistControl" is set to false which means that one clip at a time is loaded into the player (removing the playlist control next/previous clip buttons). Example 02c illustrates what happens if "allowPlaylistControl" is set to true.


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/rtmp02a.xml"
/>

"rtmp02a.xml" is defined as: 


<config>
    <openadstreamer.title>
        RTMP Example 02a - Plays non-contiguously
    </openadstreamer.title>
    <openadstreamer.json>
        {
            "allowPlaylistControl": false,

            "shows": {
                "baseURL": "rtmp://ne7c0nwbit.rtmphost.com/videoplayer",
                "streams": [ { "file":"the-black-hole.mp4", "duration":"00:00:10" } ]
            },

            "debug": {
                "debugger": "firebug",
                "levels": "fatal",
            },

            "ads": {
                "playOnce": false,
                "notice": { "textStyle": "smalltext" },
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                    }
                ],
                "schedule": [
                    {
                        "zone": "1",
                        "position": "pre-roll"
                    }
                ]
            }
       }
   </openadstreamer.json>
</config>
