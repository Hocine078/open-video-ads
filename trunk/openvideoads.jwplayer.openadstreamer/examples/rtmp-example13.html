Example 13 - Repeating Ad Positions

This example plays 3 ads sequentially 10 seconds into the main stream.


The configuration for this example is: 


<object id="ply" width="400" height="260" type="application/x-shockwave-flash" name="ply" data="../dist/4.5.swf">
  <param name="allowfullscreen" value="true"/>
  <param name="allowscriptaccess" value="always"/>
  <param name="flashvars" value="height=260&width=400&plugins=../dist/OpenAdStreamer.swf&config=config/rtmp13.xml"/>
</object>

"rtmp13.xml" is defined as: 


<config>
   <openadstreamer.title>
       RTMP Example 13
   </openadstreamer.title>
   <openadstreamer.json>
       {
            "deliveryType": "streaming",
            "allowPlaylistControl": false,

            "shows": {
            "baseURL": "rtmp://ne7c0nwbit.rtmphost.com/videoplayer",
                "streams": [
                    { "file":"the-black-hole.mp4", "duration":"00:01:00" }
                 ]
            },

            "ads": {
                 "allowAdRepetition": true,
                 "notice": { "textStyle": "smalltext" },
                 "server": {
                     "type": "OpenX",
                     "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                 },
                 "schedule": [
                     {
                        "zone": "1",
                        "position": "mid-roll",
                        "startTime": "00:00:10",
                        "repeat": 3
                     }
                 ]
            },

            "debug": {
                "levels": "fatal",
                "debugger": "firebug"
            }
        }
   </openadstreamer.json>
</config>
