Example 17 (RTMP) - Play ads only once

In this example the pre-roll ad is played only once. Once played, it is marked as "played" in the playlist so a replay of the stream does not result in the ad being replayed.


The configuration for this example is: 


<object id="ply" width="450" height="300" type="application/x-shockwave-flash" name="ply" data="../dist/4.5.swf">
  <param name="allowfullscreen" value="true"/>
  <param name="allowscriptaccess" value="always"/>
  <param name="flashvars" value="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/rtmp17.xml"/>
</object>

"rtmp17.xml" is defined as: 


<config>
   <openadstreamer.title>
      RTMP Example 17
   </openadstreamer.title>
   <openadstreamer.json>
        {
            "deliveryType": "streaming",
            "allowPlaylistControl": false,

            "shows": {
            "baseURL": "rtmp://ne7c0nwbit.rtmphost.com/videoplayer",
                 "streams": [ { "file":"the-black-hole.mp4", "duration":"00:00:10" } ]
            },

            "ads": {
                 "deliveryType": "streaming",
                 "playOnce": true,
                 "notice": { "textStyle": "smalltext" },
                 "server": {
                    "type": "OpenX",
                    "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                 },
                 "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll"
                    }
                 ]
            },

            "debug": {
                 "levels": "fatal",
                 "debugger": "firebug"
            }
        }
   </openadstreamer.json>
</config>
