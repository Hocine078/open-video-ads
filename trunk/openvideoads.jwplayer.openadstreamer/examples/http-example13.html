Example 13 (HTTP) - Repeating Ad Positions

This example plays 3 ads sequentially 10 seconds into the main stream.


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/http13.xml"
/>

"http.xml" is defined as: 


<config>
   <openadstreamer.title>
       HTTP Example 13
   </openadstreamer.title>
   <openadstreamer.json>
       {
           "allowPlaylistControl": false,

            "shows": {
                "baseURL": "http://streaming.openvideoads.org:81/shows",
                "streams": [
                    { "file":"the-black-hole.mp4", "duration":"00:01:00" }
                ]
            },

            "ads": {
               "allowAdRepetition": true,
               "notice": { "textStyle": "smalltext" },
               "servers": [
                   {
                      "type": "OpenX",
                      "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php",
                      "allowAdRepetition": true
                   }
               ],
               "schedule": [
                   {
                      "zone": "18",
                      "position": "mid-roll",
                      "startTime": "00:00:10",
                      "repeat": 3
                   }
               ]
            },

            "debug": {
               "levels": "fatal",
               "debugger": "firebug"
            }
       }
   </openadstreamer.json>
</config>
