Example 2b - A pre-roll ad and show stream - Play Once

This example plays a HTTP pre-roll ad before the main show stream. The ad slot is defined as zone "18" in our OpenX instance. Once the pre-roll ad has been played it will be removed from the playlist - it cannot be replayed unless the player/page is refreshed.




The configuration for this example is: 


<object id="ply" width="450" height="300" type="application/x-shockwave-flash" name="ply" data="../dist/4.5.swf">
  <param name="allowfullscreen" value="true"/>
  <param name="allowscriptaccess" value="always"/>
  <param name="flashvars" value="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/http02b.xml"/>
</object>

"http02b.xml" is defined as: 


<config>
    <openadstreamer.title>
        HTTP Example 02b - Play Once
    </openadstreamer.title>
    <openadstreamer.json>
        {
            "deliveryType": "progressive",
            "allowPlaylistControl": false,

             "shows": {
                 "streams": [
                      {
                          "file":"http://streaming.openvideoads.org:81/shows/the-black-hole.mp4",
                          "duration":"00:00:30"
                      }
                 ]
             },

             "debug": {
                 "debugger": "firebug",
                 "levels": "fatal",
             },

             "ads": {
                 "playOnce": true,
                 "notice": { "textStyle": "smalltext" },
                 "server": {
                     "type": "OpenX",
                     "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                 },
                 "schedule": [
                     {
                         "zone": "18",
                         "position": "pre-roll"
                     }
                 ]
             }
        }
    </openadstreamer.json>
</config>
