Example 2c - A pre-roll ad and show stream - Playlist Control

This example plays a RTMP pre-roll ad before the main show stream. The ad slot is defined as zone "1" in our OpenX instance. The ad is repeated every time the stream is replayed. "allowPlaylistControl" is set to true which means that the clips are loaded into the player as 1 playlist and the playlist control buttons are enabled (skip to the next clip/back). Example 02a illustrates what happens if "allowPlaylistControl" is set to false.


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/rtmp02c.xml"
/>

"rtmp02c.xml" is defined as: 


<config>
   <openadstreamer.title>
       RTMP Example 02c - Plays Contiguously
   </openadstreamer.title>
   <openadstreamer.json>
       {
           "allowPlaylistControl": true,

           "shows": {
               "baseURL": "rtmp://ne7c0nwbit.rtmphost.com/videoplayer",
               "streams": [
                    {
                         "file":"the-black-hole.mp4",
                         "duration":"00:00:10"
                    }
               ]
           },

           "debug": {
               "debugger": "firebug",
               "levels": "fatal",
           },

           "ads": {
               "playOnce": false,
               "notice": { "textStyle": "smalltext" },
               "server": {
                   "type": "OpenX",
                   "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
               },
               "schedule": [
                   {
                       "zone": "1",
                       "position": "pre-roll"
                   }
               ]
           }
       }
  </openadstreamer.json>
</config>
