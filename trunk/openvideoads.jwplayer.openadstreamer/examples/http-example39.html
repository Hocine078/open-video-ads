Example 39 - Configuring Pseudo-Streaming Providers

The following example illustrates how to configure a different pseudo-streaming provider. By default, 'lighttpd' is configured as the streamer of choice for HTTP pseudo-streaming delivery.


The configuration for this example is: 


<embed 
  id="ply"
  data="../dist/4.5.swf"
  src="../dist/4.5.swf" 
  width="450"
  height="300"
  allowscriptaccess="always"
  allowfullscreen="true"
  flashvars="height=300&width=450&plugins=../dist/OpenAdStreamer.swf&config=config/http39.xml"
/>

"http39.xml" is defined as: 


<config>
   <openadstreamer.title>
       HTTP Example 39 - Supporting an XMOOV pseudo-streaming provider
   </openadstreamer.title>
   <openadstreamer.json>
       {
          "allowPlaylistControl": false,
          "shows": {
              "providers" : {
                   "http": "http://static.openvideoads.org/xmoov/xmoov.php"
              },
              "streams": [
                   {
                        "file": "dogs_600_with_metadata.flv",
                        "duration": "00:00:19"
                   }
              ]
          },

          "debug": {
              "debugger": "firebug",
              "levels": "fatal, config, segment_formation, playlist",
          },

          "ads": {
              "playOnce": false,
              "notice": { "textStyle": "smalltext" },
              "server": {
                  "type": "OpenX",
                  "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
              },
              "schedule": [
                  {
                     "zone": "18",
                     "position": "pre-roll"
                  }
              ]
          }
       }
   </openadstreamer.json>
</config>
