Example 17 (HTTP) - Play ads only once

In this example the pre-roll ad is played only once. Once played, it is marked as "played" in the playlist so a replay of the stream does not result in the ad being replayed.

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.5.swf", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "shows": {
                "baseURL": "http://streaming.openvideoads.org/shows",
                "streams": [
                    { "file":"the-black-hole.mp4", "duration":"00:00:20" }
                ]
            },

            "ads": {
                "playOnce": true,
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                    }
                ],
                "schedule": [
                    {
                       "zone": "5",
                       "position": "pre-roll"
                    }
                ],
            },

            "debug": {
                "levels": "fatal",
                "debugger": "firebug"
            }
        }
    }
});
