Example 17 (HTTP) - Play ads only once

In this example the pre-roll ad is played only once. Once played, it is marked as "played" in the playlist so a replay of the stream does not result in the ad being replayed.

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.3-dev.swf", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "deliveryType": "progressive",

            "shows": {
                "baseURL": "http://static.bouncingminds.com/shows",
                "streams": [
                    { "file":"mp4:bbb_640x360.mp4", "duration":"00:00:20" }
                ]
            },

            "ads": {
                "deliveryType": "streaming",
                "playOnce": true,
                "server": {
                    "type": "OpenX",
                    "vastURL": "http://openx.bouncingminds.com/openx/www/delivery/fc.php"
                },
                "schedule": [
                    {
                       "zone": "5",
                       "position": "pre-roll"
                    }
                ],
            },

            "debug": {
                "levels": "all",
                "debugger": "firebug"
            }
        }
    }
});
