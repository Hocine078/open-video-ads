Example 8 (HTTP) - Changing the "This is an advert" message

This example illustrates how the "this is an advert" type message can be turned on and off, and the text changed. For this example, the text has been set to "My New Ad Notice - _seconds_ seconds".

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.4.swf", {
   plugins: {
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "overlays": {
                "regions": [
                    {
                        "id": "my-ad-notice",
                        "verticalAlign": "bottom",
                        "horizontalAlign": "right",
                        "backgroundColor": "transparent",
                        "width": "100pct",
                        "height": 40,
                        "style": ".smalltext { font-style: italic; font-size:10; }"
                    }
                ]
            },

            "ads": {
                "server": {
                    "type": "OpenX",
                    "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                },

                "schedule": [
                    {
                        "zone": "5",
                        "position": "pre-roll",
                        "notice": {
                            "show": true,
                            "region": "my-ad-notice",
                            "message": "<p class=\"smalltext\" 
                                        align=\"right\">My New Ad Notice - 
                                        _seconds_ seconds</p>"
                        },
                    }
                ]
            },

            "debug": {
                "levels": "fatal",
                "debugger": "firebug, demonster"
            }
        }
    }
});
