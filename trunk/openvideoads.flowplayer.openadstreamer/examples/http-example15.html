Example 15 (HTTP) - Selection of media based on bitrate

This example illustrates how to specify a 'bitrate' as a selection criteria for the mediafiles returned by the openX ad server.

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.3-dev.swf", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "deliveryType": "progressive",

            "shows": {
                "baseURL": "http://static.bouncingminds.com/shows",
                "streams": [
                    { "file":"mp4:bbb_640x360.mp4", "duration":"00:01:00" }
                ]
            },

            "ads": {
                "bitrate": "500-700",
                "server": {
                    "type": "OpenX",
                    "vastURL": "http://openx.bouncingminds.com/openx/www/delivery/fc.php"
                },
                "schedule": [
                    {
                        "zone": "5",
                        "position": "pre-roll"
                    }
                ]
            },

            "debug": {
                "levels": "all",
                "debugger": "firebug"
            }
        }
    }
});
