Example 16 (HTTP) - Overlay with 'click to website'

Click on the overlay to be redirected to the advertiser website.

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.3.swf", {
    plugins: {
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "deliveryType": "progressive",

            "shows": {
                "baseURL": "http://streaming.openvideoads.org/shows",
                "streams": [
                    { "file":"the-black-hole.mp4", "duration":"00:00:20" }
                ]
            },

            "ads": {
                "server": {
                    "type": "OpenX",
                    "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                },
                "schedule": [
                    {
                        "zone": "30",
                        "startTime": "00:00:01",
                        "width": 450,
                        "height": 50
                    }
                ]
            },

            "debug": {
                "levels": "fatal",
                "debugger": "firebug"
            }
        }
    }
});
