Example 12 - Selection of video media based on mime type

This example illustrate how to restrict video selection based on mime-type - 'video/x-flv' in this case.

The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.3.swf", {
    plugins: {
        rtmp: {
            url: "../dist/flowplayer.rtmp-3.1.2.swf"
        },
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "deliveryType": "streaming",
            "streamType": "flv",
            "metaData": false,

            "ads": {
                "server": {
                   "type": "OpenX",
                   "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                },
                "schedule": [
                    {
                       "zone": "13",
                       "position": "pre-roll"
                    }
                ]
            },

            "debug": {
                "levels": "fatal",
                "debugger": "firebug"
            }
        }
    }
});
