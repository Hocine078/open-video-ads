Example 2 (RTMP) - A pre-roll ad with a show stream



The configuration for this example is: 


flowplayer("a.example", "../dist/flowplayer-3.1.3.swf", {
    plugins: {
        rtmp: {
            url: "../dist/flowplayer.rtmp-3.1.2.swf"
        },
        controls: {
            autoHide: "always"
        },

        openAdStreamer: {
            url: '../dist/OpenAdStreamer.swf',

            "deliveryType": "streaming",

            "shows": {
                "baseURL": "rtmp://ne7c0nwbit.rtmphost.com/videoplayer",
                "streams": [ { "file":"mp4:the-black-hole.mp4", "duration":"00:00:10" } ]
            },

            "debug": {
                "debugger": "firebug",
                "levels": "fatal",
            },

            "ads": {
                "playOnce": false,
                "server": {
                    "type": "OpenX",
                    "vastURL": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                },
                "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll"
                    }
                ]
            }
        }
    }
});
